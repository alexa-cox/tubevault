import PropTypes from 'prop-types';
import React, { memo } from 'react';
import { useNavigationHandler } from '../../hooks';
import { ROLES, TAB_INDEXES, CSS_CLASSES } from '../../utils';

/**
 * @typedef {Object} LogoStyles
 * @property {string} logoWrapper - Container class for the logo
 * @property {string} logo - Class for the logo image
 */

/** @type {LogoStyles} */
import styles from './logo.module.css';

function Logo({ width = 50, height = width, className }) {
  const { handleClick, handleKeyPress } = useNavigationHandler();

  return (
    <div
      className={styles[CSS_CLASSES.LOGO.WRAPPER]}
      role={ROLES.BUTTON}
      tabIndex={TAB_INDEXES.FOCUSABLE}
      onClick={handleClick}
      onKeyDown={handleKeyPress}
      aria-label='Return to homepage'
    >
      <img
        style={{ width, height }}
        className={`${className} ${styles[CSS_CLASSES.LOGO.IMAGE]}`}
        src='/images/logo.svg'
        alt='Tube Valut Logo'
      />
    </div>
  );
}

Logo.propTypes = {
  width: PropTypes.number,
  height: PropTypes.number,
  className: PropTypes.string,
};

export default memo(Logo);
